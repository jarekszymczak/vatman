def show_analysis(marz_item, offset):
    for rss in recall_set_stems:
        try:
            index = marz_item.index(rss)
            break
        except:
            index = 0
    min_boundary = max(0, index-offset)
    max_boundary = min(len(marz_item), index+offset)
    if marz_item[min_boundary:max_boundary]:
        return index / len(marz_item), marz_item[min_boundary:max_boundary]
    else:
        return 1.0, ''
